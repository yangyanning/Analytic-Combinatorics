$\rhd$ \textbf{I.2.} \textit{Unimodal permutations}.
\begin{equation}
    \mathcal{P}^{\mathcal{U}} = \mathrm{S}\textsc{et}\left(\mathcal{Z}\right) \star \mathcal{Z}^{\blacksquare} \star \mathrm{S}\textsc{et}\left(\mathcal{Z}\right)
\end{equation}
\begin{equation}
    P^{U}\left(z\right) = \int_{0}^{z}\left(e^{t}\cdot \frac{d}{dt}t\cdot e^{t}\right)dt = \frac{1}{2}\left(e^{2z} - 1\right)
\end{equation}
which yields
\begin{equation}
    P^{U}_{n} = n!\frac{1}{2}\left[z^{n}\right]e^{2z} = n!\frac{1}{2}\frac{2^{n}}{n!} = 2^{n - 1}
\end{equation}
Alternatively,
\begin{equation}
    P^{U}_{n} = \sum_{k = 1}^{n}\binom{n - 1}{k - 1} = \sum_{j = 0}^{n - 1}\binom{n - 1}{j} = 2^{n - 1}
\end{equation}
where $\displaystyle k$ is the position of $\displaystyle n$ in the permutation.

$\rhd$ \textbf{I.16.} \textit{Partitions with summands bounded in number and size}. The number of partitions
of size $\displaystyle n$ with at most $\displaystyle k$ summands, each at most $\displaystyle l$, is
\begin{equation} \label{eq:I.16.1}
    \left[z^{n}\right]
    \frac{\left(1 - z\right)\left(1 - z^{2}\right)\cdots \left(1 - z^{k + l}\right)}
         {\left(\left(1 - z\right)\left(1 - z^{2}\right)\cdots \left(1 - z^{k}\right)\right)\cdot \left(\left(1 - z\right)\left(1 - z^{2}\right)\cdots \left(1 - z^{l}\right)\right)}
\end{equation}
Let $\displaystyle f\left(k, l\right)$ be the generating function in which the coefficient of $\displaystyle z^{n}$ is the number
of partitions of size $\displaystyle n$ with at most $\displaystyle k$ summands, each at most $\displaystyle l$.
Consider the largest possible part in our partition. There are two mutually exclusive cases:
the largest part is $\displaystyle < l$, and the largest part is $\displaystyle = l$. This yields the functional equation:
\begin{equation} \label{eq:I.16.2}
    f\left(k, l\right) = f\left(k, l - 1\right) + z^{l}f\left(k - 1, l\right)
\end{equation}
We can verify that \ref{eq:I.16.1} satisfies \ref{eq:I.16.2}. Can we derive \ref{eq:I.16.1} from \ref{eq:I.16.2}?
Let $\displaystyle F_{k}\left(x\right)$ be the generating function for $\displaystyle f\left(k, l\right)$ with respect to the index $\displaystyle l$:
\begin{equation}
    F_{k}\left(x\right) = \sum_{l = 0}^{\infty}f\left(k, l\right)x^{l}
\end{equation}
For $\displaystyle k = 0$, there is only one partition of $0$ into at most $0$ parts (the empty partition).
Thus $\displaystyle f\left(0,l\right) = 1$ for all $\displaystyle l\geq 0$.
\begin{equation}
    F_{0}\left(x\right) = \sum_{l = 0}^{\infty}1\cdot x^{l} = \frac{1}{1 - x}
\end{equation}
Take \ref{eq:I.16.2}, multiply by $\displaystyle x^{l}$, and sum over $\displaystyle l\geq 1$:
\begin{equation}
    \sum_{l = 1}^{\infty}f\left(k, l\right)x^{l} - \sum_{l = 1}^{\infty}f\left(k, l - 1\right)x^{l} =
        \sum_{l = 1}^{\infty}f\left(k - 1, l\right)\left(zx\right)^{l}
\end{equation}
\begin{itemize}
    \item The first term is $\displaystyle F_{k}\left(x\right) - f\left(k, 0\right)$. Since $\displaystyle f\left(k, 0\right) = 1$ (at most $0$ size summands),
    this equals $\displaystyle F_{k}\left(x\right) - 1$.
    \item The second term is $\displaystyle x\sum_{j = 0}^{\infty}f\left(k, j\right)x^{j} = xF_{k}\left(x\right)$.
    \item The right-hand side is $\displaystyle F_{k -1}\left(zx\right) - f\left(k - 1, 0\right) = F_{k - 1}\left(zx\right) - 1$.
\end{itemize}
Plugging these in:
\begin{equation}
    \left(1 - x\right)F_{k}\left(x\right) - 1 = F_{k - 1}\left(zx\right) - 1
        \Longrightarrow F_{k}\left(x\right) = \frac{1}{1 - x}F_{k - 1}\left(zx\right)
\end{equation}
Now we apply this functional relation repeatedly:
\begin{equation}
    \begin{aligned}
        F_{k}\left(x\right)
        &= \frac{1}{1 - x}F_{k - 1}\left(zx\right) \\
        &= \frac{1}{1 - x}\cdot \frac{1}{1 - zx}F_{k - 2}\left(z^{2}x\right) \\
        &= \frac{1}{\left(1 - x\right)\left(1 - zx\right)\cdots \left(1 - z^{k - 1}x\right)}F_{0}\left(z^{k}x\right)
    \end{aligned}
\end{equation}
Since $\displaystyle F_{0}\left(x\right) = \frac{1}{1 - x}$, we have $\displaystyle F_{0}\left(z^{k}x\right) = \frac{1}{1 - z^{k}x}$. Thus:
\begin{equation}
F_{k}\left(x\right) = \sum_{l = 0}^{\infty}f\left(k, l\right)x^{l} = \frac{1}{\left(1 - x\right)\left(1 - zx\right)\cdots \left(1 - z^{k}x\right)}
\end{equation}
Since all the roots of the denominator are distinct (assuming $\displaystyle |z| < 1$ or that $\displaystyle z$ is a formal variable),
we can decompose this into a sum of partial fractions:
\begin{equation}
    \frac{1}{\prod_{i = 0}^{k}\left(1 - z^{i}x\right)} = \sum_{j = 0}^{k}\frac{A_{j}}{1 - z^{j}x}
\end{equation}
To find a specific $\displaystyle A_{j}$, we multiply both sides by $\displaystyle \left(1 - z^{j}x\right)$ and then
set $\displaystyle x = z^{-j}$:
\begin{equation}
    \begin{aligned}
        A_{j}
        &= \left[\frac{1 - z^{j}x}{\prod_{i = 0}^{k}\left(1 - z^{i}x\right)}\right]_{x = z^{-j}} \\
        &= \frac{1}{\prod_{i = 0}^{j - 1}\left(1 - z^{i}z^{-j}\right)\prod_{i = j + 1}^{k}\left(1 - z^{i}z^{-j}\right)} \\
        &= \frac{\left(-1\right)^{j}z^{j\left(j + 1\right)/2}}{\prod_{i = 1}^{j}\left(1 - z^{i}\right)\prod_{i = 1}^{k - j}\left(1 - z^{i}\right)}
    \end{aligned}
\end{equation}
The coefficient of $\displaystyle x^{l}$ in $\displaystyle F_{k}\left(x\right)$ is the sum of the coefficients of $\displaystyle x^{l}$
in each partial fraction. For a term $\displaystyle \frac{A_{j}}{1 - z^{j}x}$, the coefficient is $\displaystyle A_{j}\left(z^{j}\right)^{l}$.
\begin{equation}
    \begin{aligned}
        f\left(k, l\right)
        &= \sum_{j = 0}^{k}A_{j}\left(z^{j}\right)^{l} \\
        &= \sum_{j = 0}^{k}\frac{\left(-1\right)^{j}z^{j\left(j + 1\right)/2}\cdot z^{jl}}
                                {\prod_{i = 1}^{j}\left(1 - z^{i}\right)\prod_{i = 1}^{k - j}\left(1 - z^{i}\right)} \\
        &= \frac{1}{\prod_{i = 1}^{k}\left(1 - z^{i}\right)}\sum_{j = 0}^{k}\binom{k}{j}_{z}\left(-1\right)^{j}z^{j\left(j + 1\right)/2 + jl}
    \end{aligned}
\end{equation}
where $\displaystyle \binom{k}{j}_{z}$ is the Gaussian binomial coefficient:
\begin{equation}
    \begin{aligned}
        \binom{k}{j}_{z}
        &= \frac{\prod_{i = 1}^{k}\left(1 - z^{i}\right)}
                {\prod_{i = 1}^{j}\left(1 - z^{i}\right)\prod_{i = 1}^{k - j}\left(1 - z^{i}\right)} \\
        &= \frac{\left(1 - z^{k}\right)\left(1 - z^{k - 1}\right)\cdots \left(1 - z^{k - j + 1}\right)}
                {\left(1 - z\right)\left(1 - z^{2}\right)\cdots \left(1 - z^{j}\right)}
    \end{aligned}
\end{equation}
We can rewrite the exponent in the sum:
\begin{equation}
    \frac{j\left(j + 1\right)}{2} + jl = \frac{j\left(j - 1\right)}{2} + j\left(l + 1\right)
\end{equation}
Now the sum becomes
\begin{equation}
    f\left(k, l\right)
    = \frac{1}{\prod_{i = 1}^{k}\left(1 - z^{i}\right)}\sum_{j = 0}^{k}\binom{k}{j}_{z}\left(-1\right)^{j}z^{j\left(j - 1\right)/2}\left(z^{l + 1}\right)^{j}
\end{equation}
The $q$-binomial theorem states that for a variable $X$:
\begin{equation} \label{eq:I.16.3}
    \prod_{i = 0}^{k - 1}\left(1 - z^{i}X\right) = \sum_{j = 0}^{k}\binom{k}{j}_{z}\left(-1\right)^{j}z^{j\left(j - 1\right)/2}X^{j}
\end{equation}
Plugging $\displaystyle X = z^{l + 1}$ into the theorem yields \ref{eq:I.16.1}:
\begin{equation}
    \begin{aligned}
        f\left(k, l\right)
        &= \frac{\left(1 - z^{l + 1}\right)\left(1 - z^{l + 2}\right)\cdots \left(1 - z^{l + k}\right)}
                {\left(1 - z^{1}\right)\left(1 - z^{2}\right)\cdots \left(1 - z^{k}\right)} \\
        &= \frac{\left(1 - z\right)\left(1 - z^{2}\right)\cdots \left(1 - z^{k + l}\right)}
                {\left(\left(1 - z\right)\left(1 - z^{2}\right)\cdots \left(1 - z^{k}\right)\right)\cdot \left(\left(1 - z\right)\left(1 - z^{2}\right)\cdots \left(1 - z^{l}\right)\right)}
    \end{aligned}
\end{equation}
To prove \ref{eq:I.16.3}, let $\displaystyle P_{k}\left(X\right)$ be the product we are expanding:
\begin{equation}
    P_{k}\left(X\right) = \prod_{i = 0}^{k - 1}\left(1 - z^{i}X\right)
\end{equation}
We want to show that $\displaystyle P_{k}\left(X\right) = \sum_{j = 0}^{k}\binom{k}{j}_{z}\left(-1\right)^{j}z^{j\left(j - 1\right)/2}X^{j}$.
Consider the relationship between $\displaystyle P_{k}\left(X\right)$ and $\displaystyle P_{k - 1}\left(X\right)$. By definition:
\begin{equation}
    P_{k}\left(X\right) = \left(1 - z^{k - 1}X\right)P_{k - 1}\left(X\right)
\end{equation}
Now notice what happens if we replace $\displaystyle X$ with $\displaystyle zX$ in $\displaystyle P_{k}\left(X\right)$:
\begin{equation}
    P_{k}\left(zX\right) = \prod_{i = 0}^{k - 1}\left(1 - z^{i + 1}X\right) = \frac{1 - z^{k}X}{1 - X}P_{k}\left(X\right)
\end{equation}
Rearranging this gives us a functional equation for $\displaystyle P_{k}\left(X\right)$:
\begin{equation}
    \left(1 - X\right)P_{k}\left(zX\right) = \left(1 - z^{k}X\right)P_{k}\left(X\right)
\end{equation}
Let $\displaystyle P_{k}\left(X\right) = \sum_{j = 0}^{k}c_{j}X^{j}$. We substitute this power series into the functional equation:
\begin{equation}
    \left(1 - X\right)\sum_{j = 0}^{k}c_{j}\left(zX\right)^{j} = \left(1 - z^{k}X\right)\sum_{j = 0}^{k}c_{j}X^{j}
\end{equation}
Now, we compare the coefficients of $\displaystyle X^{j}$ on both sides. For a specific power $\displaystyle j$:
\begin{equation}
    c_{j}z^{j} - c_{j - 1}z^{j - 1} = c_{j} - c_{j - 1}z^{k}
\end{equation}
Rearranging to solve for $\displaystyle c_{j}$ in terms of $\displaystyle c_{j - 1}$:
\begin{equation}
    c_{j} = c_{j - 1}\frac{z^{j - 1} - z^{k}}{z^{j} - 1} = -c_{j - 1}z^{j - 1}\frac{1 - z^{k - j + 1}}{1 - z^{j}}
\end{equation}
Starting from the base case $\displaystyle c_{0} = P_{k}\left(0\right) = 1$, we can find $\displaystyle c_{j}$ by iteration:
\begin{equation}
    c_{1} = -c_{0}z^{0}\frac{1 - z^{k}}{1 - z^{1}}, \quad
    c_{2} = -c_{1}z^{1}\frac{1 - z^{k - 1}}{1 - z^{2}}
        = \left(-1\right)^2z^{0 + 1}\frac{\left(1 - z^{k}\right)\left(1 - z^{k - 1}\right)}{\left(1 - z^{1}\right)\left(1 - z^{2}\right)}
\end{equation}
The exponent of $\displaystyle z$ becomes $\displaystyle 0 + 1 + 2 + \cdots + \left(j - 1\right) = \frac{j\left(j - 1\right)}{2}$.
The fraction is the definition of the Gaussian binomial coefficient $\displaystyle \binom{k}{j}_{z}$.
Thus, the coefficient is:
\begin{equation}
    c_{j} = \left(-1\right)^{j}z^{j\left(j - 1\right)/2}\binom{k}{j}_{z}
\end{equation}
Plugging the coefficients back into the sum, we obtain:
\begin{equation}
    \prod_{i = 0}^{k - 1}\left(1 - z^{i}X\right) = \sum_{j = 0}^{k}\binom{k}{j}_{z}\left(-1\right)^{j}z^{j\left(j - 1\right)/2}X^{j}
\end{equation}
This completes the proof.

$\rhd$ \textbf{I.17.} \textit{Systems of linear Diophantine inequalities}.
\begin{equation}
    \left\{x_{1}\geq 0,\; x_{2}\geq 2x_{1},\; x_{3}\geq 2x_{2},\; x_{4}\geq 2x_{3}\right\}
\end{equation}
We introduce non-negative slack variables $\displaystyle y_{j}\in \mathbb{Z}_{\geq 0}$
that measure the excess of each term over its minimum bound:
\begin{align}
    x_{1} &= y_{1} \\
    x_{2} &= 2x_{1} + y_{2} = 2y_{1} + y_{2} \\
    x_{3} &= 2x_{2} + y_{3} = 4y_{1} + 2y_{2} + y_{3} \\
    x_{4} &= 2x_{3} + y_{4} = 8y_{1} + 4y_{2} + 2y_{3} + y_{4}
\end{align}
The size of the composition is $\displaystyle S = x_{1} + x_{2} + x_{3} + x_{4} = 15y_{1} + 7y_{2} + 3y_{3} + y_{4}$.
Since the variables $\displaystyle y_{1}, y_{2}, y_{3}, y_{4}$ are independent and each ranges freely over the non-negative integers,
the OGF for the class $\displaystyle \mathcal{F}$ is the product of the generating functions for the individual $\displaystyle y_{j}$:
\begin{itemize}
    \item For $\displaystyle 15y_{1}: \sum_{y_{1} = 0}^{\infty}\left(z^{15}\right)^{y_{1}} = \frac{1}{1 - z^{15}}$
    \item For $\displaystyle  7y_{2}: \sum_{y_{2} = 0}^{\infty}\left(z^{ 7}\right)^{y_{2}} = \frac{1}{1 - z^{ 7}}$
    \item For $\displaystyle  3y_{3}: \sum_{y_{3} = 0}^{\infty}\left(z^{ 3}\right)^{y_{3}} = \frac{1}{1 - z^{ 3}}$
    \item For $\displaystyle   y_{4}: \sum_{y_{4} = 0}^{\infty}\left(z^{ 1}\right)^{y_{4}} = \frac{1}{1 - z^{ 1}}$
\end{itemize}
Multiplying these together yields the desired OGF:
\begin{equation}
    F\left(z\right) = \frac{1}{\left(1 -z\right)\left(1 -z^{3}\right)\left(1 -z^{7}\right)\left(1 -z^{15}\right)}
\end{equation}

\vspace{\baselineskip}
\begin{equation}
    \left\{x_{1} + x_{2}\leq x_{3}\right\}
\end{equation}
Following the same approach as in the previous problem, we use slack variables to convert the inequality into an equality of independent parts.
The constraint $\displaystyle x_{3}\geq x_{1} + x_{2}$ can be rewritten by introducing a non-negative integer $\displaystyle y\geq 0$:
\begin{equation}
    x_{3} = x_{1} + x_{2} + y
\end{equation}
The size of the composition is $\displaystyle S = x_{1} + x_{2} + x_{3} + x_{4} = 2x_{1} + 2x_{2} + y + x_{4}$.
Since the variables $\displaystyle x_{1}, x_{2}, y, x_{4}$ are independent and each ranges freely over the non-negative integers,
the OGF is the product of the generating functions for the individual terms in the sum:
\begin{itemize}
    \item For $\displaystyle 2x_{1}: \frac{1}{1 - z^{2}}$
    \item For $\displaystyle 2x_{2}: \frac{1}{1 - z^{2}}$
    \item For $\displaystyle y: \frac{1}{1 - z}$
    \item For $\displaystyle x_{4}: \frac{1}{1 - z}$
\end{itemize}
Multiplying these together gives:
\begin{equation}
    F\left(z\right) = \frac{1}{\left(1 -z^{2}\right)^{2}\left(1 -z\right)^{2}}
\end{equation}

\vspace{\baselineskip}
\begin{equation}
    \left\{x_{1} + x_{2}\geq x_{3}\right\}
\end{equation}
Because the inequality does not immediately suggest a single slack variable to absorb the excess,
the most robust approach is to use the complement method.
We know the OGF for four unrestricted variables $\displaystyle \left(x_{1},x_{2},x_{3},x_{4}\right)$
with $\displaystyle x_{i}\geq 0$ is:
\begin{equation}
    U\left(z\right) = \frac{1}{\left(1 - z\right)^{4}}
\end{equation}
The set of all such compositions partitions naturally into two disjoint subsets: those satisfying
$\displaystyle x_{1} + x_{2} < x_{3}$ (i.e., $\displaystyle x_{1} + x_{2} + 1\leq x_{3}$), and those satisfying
$\displaystyle x_{1} + x_{2}\geq x_{3}$ (our target class). We find the OGF for the complement
$\displaystyle x_{1} + x_{2} < x_{3}$. Set $\displaystyle x_{3} = x_{1} + x_{2} + 1 + y$, where $\displaystyle y\geq 0$.
The total sum $\displaystyle S$ is:
\begin{equation}
    S = x_{1} + x_{2} + \left(x_{1} + x_{2} + 1 + y\right) + x_{4} = 2x_{1} + 2x_{2} + y + x_{4} + 1
\end{equation}
The OGF for this complement, which we denote by $\displaystyle C\left(z\right)$, is:
\begin{equation}
    C\left(z\right) = \frac{z}{\left(1 - z^{2}\right)^{2}\left(1 - z\right)^2}
\end{equation}
The OGF for our constraint is then:
\begin{equation}
    F\left(z\right)
        = U\left(z\right) - C\left(z\right)
        = \frac{1}{\left(1 - z\right)^{4}} - \frac{z}{\left(1 - z^{2}\right)^{2}\left(1 - z\right)^2}
        = \frac{1 + z + z^{2}}{\left(1 - z\right)^{4}\left(1 + z\right)^{2}}
\end{equation}

\vspace{\baselineskip}
\begin{equation}
    \left\{x_{1} + x_{2}\leq x_{3} + x_{4}\right\}
\end{equation}
To find the OGF for $\displaystyle \left\{x_{1} + x_{2}\leq x_{3} + x_{4}\right\}$, we can use the complement method
combined with the symmetry of the variables. Let $\displaystyle A = x_{1} + x_{2}$ and $\displaystyle B = x_{3} + x_{4}$.
We seek the OGF for compositions satisfying $\displaystyle A\leq B$. In the universe of all possible compositions
$\displaystyle \left(x_{1}, x_{2}, x_{3}, x_{4}\right)$, there are three mutually exclusive cases:
$\displaystyle 1.\; A < B,\; 2.\; A > B,\; 3.\; A = B$. By symmetry, the OGF for case 1 must be identical to the OGF for case 2,
since $\displaystyle A$ and $\displaystyle B$ are built from the same number of variables with the same weights.
Denote this common OGF by $\displaystyle G_{<}\left(z\right)$, and let $\displaystyle G_{=}\left(z\right)$ be the OGF for case 3.
The total unrestricted OGF is $\displaystyle U\left(z\right) = \frac{1}{\left(1 - z\right)^{4}}$. Therefore:
\begin{equation}
    U\left(z\right) = 2G_{<}\left(z\right) + G_{=}\left(z\right)
\end{equation}
Our target OGF is $\displaystyle F\left(z\right) = G_{<}\left(z\right) + G_{=}\left(z\right)$. Solving the system gives:
\begin{equation}
    F\left(z\right) = \frac{U\left(z\right) + G_{=}\left(z\right)}{2}
\end{equation}
Now $\displaystyle G_{=}\left(z\right)$ is the OGF for compositions in which the sum of the first two variables equals
the sum of the last two.
\begin{itemize}
    \item The number of compositions with $\displaystyle x_{1} + x_{2} = n$ is
        $\displaystyle \left[z^{n}\right]\frac{1}{\left(1 - z\right)^{2}} = n + 1$.
    \item The number of compositions with $\displaystyle x_{3} + x_{4} = n$ is the same.
    \item For the equality $\displaystyle x_{1} + x_{2} = x_{3} + x_{4}$ to hold, both sums must equal some $\displaystyle n\geq 0$.
        \begin{equation}
            G_{=}\left(z\right)
                = \sum_{n = 0}^{\infty}\left(n + 1\right)^{2}z^{2n}
                = \frac{1 + z^{2}}{\left(1 - z^{2}\right)^{3}}
        \end{equation}
\end{itemize}
Substituting $\displaystyle U\left(z\right)$ and $\displaystyle G_{=}\left(z\right)$ into our formula yields:
\begin{equation}
    F\left(z\right)
        = \frac{1}{2}\left[\frac{1}{\left(1 - z\right)^{4}} + \frac{1 + z^{2}}{\left(1 - z^{2}\right)^{3}}\right]
        = \frac{1 + z + 2z^{2}}{\left(1 - z\right)^{4}\left(1 + z\right)^{3}}
\end{equation}

\vspace{\baselineskip}
\begin{equation}
    \left\{x_{1}\leq x_{2},\; x_{2}\geq x_{3},\; x_{3}\leq x_{4}\right\}
\end{equation}
Because the inequalities alternate direction (up, down, up), we cannot use a straightforward chain of slack variable substitutions
as we did for $\displaystyle x_{1}\leq x_{2}\leq x_{3}\leq x_{4}$. Instead, we apply the inclusion–exclusion principle to the constraints.
We begin with the universe of all unrestricted compositions, whose OGF is $\displaystyle U\left(z\right) = \frac{1}{\left(1 - z\right)^{4}}$.
We wish to enforce three conditions:
\begin{itemize}
    \item $\displaystyle C_{1}: x_{1}\leq x_{2}$
    \item $\displaystyle C_{2}: x_{2}\geq x_{3}$
    \item $\displaystyle C_{3}: x_{3}\leq x_{4}$
\end{itemize}
However, it is simpler to work with chains of "less than or equal" inequalities. Consider the complement of the downward middle step.
Take the set of all compositions satisfying both $\displaystyle x_{1}\leq x_{2}$ and $\displaystyle x_{3}\leq x_{4}$.
From this set, we subtract those compositions that violate the middle constraint (i.e., $\displaystyle x_{2} < x_{3}$). \\
The OGF for the set defined by $\displaystyle x_{1}\leq x_{2}$ and $\displaystyle x_{3}\leq x_{4}$:
These two pairs are independent.
\begin{itemize}
    \item For $\displaystyle x_{1}\leq x_{2}$, set $\displaystyle x_{2} = x_{1} + y_{1}$. The total contribution is $\displaystyle 2x_{1} + y_{1}$,
        so the OGF is $\displaystyle \frac{1}{\left(1 - z^{2}\right)\left(1 - z\right)}$.
    \item For $\displaystyle x_{3}\leq x_{4}$, the OGF is identically $\displaystyle \frac{1}{\left(1 - z^{2}\right)\left(1 - z\right)}$.
\end{itemize}
Since the two pairs are independent, we multiply their OGFs:
\begin{equation}
    A\left(z\right) = \frac{1}{\left(1 - z^{2}\right)^{2}\left(1 - z\right)^{2}}
\end{equation}
Now we find the OGF for the violation $\displaystyle x_{1}\leq x_{2} < x_{3}\leq x_{4}$ by introducing slack variables:
\begin{itemize}
    \item $\displaystyle x_{1} = y_{1}$
    \item $\displaystyle x_{2} = y_{1} + y_{2}$
    \item $\displaystyle x_{3} = y_{1} + y_{2} + 1 + y_{3}$
    \item $\displaystyle x_{4} = y_{1} + y_{2} + 1 + y_{3} + y_{4}$
\end{itemize}
Summing the four expressions gives:
\begin{equation}
    S = 4y_{1} + 3y_{2} + 2y_{3} + y_{4} + 2
\end{equation}
The OGF for this violation is therefore:
\begin{equation}
    V\left(z\right) = \frac{z^{2}}{\left(1 - z^{4}\right)\left(1 - z^{3}\right)\left(1 - z^{2}\right)\left(1 - z\right)}
\end{equation}
The OGF for the zigzag constraint is the OGF for the two independent pairs minus the OGF for the cases where the peak at $\displaystyle x_{2}$ fails:
\begin{equation}
    \begin{aligned}
        F\left(z\right)
        &= \frac{1}{\left(1 - z^{2}\right)^{2}\left(1 - z\right)^{2}} - \frac{z^{2}}{\left(1 - z^{4}\right)\left(1 - z^{3}\right)\left(1 - z^{2}\right)\left(1 - z\right)} \\
        &= \frac{1}{\left(1 - z\right)\left(1 - z^{2}\right)}\left[\frac{1}{\left(1 - z\right)\left(1 - z^{2}\right)} - \frac{z^{2}}{\left(1 - z^{3}\right)\left(1 - z^{4}\right)}\right]
    \end{aligned}
\end{equation}

$\rhd$ \textbf{I.18.} \textit{Odd versus distinct summands}.
We establish a one-to-one correspondence between partitions into odd parts and partitions into distinct parts.
\begin{itemize}
    \item Odd $\rightarrow$ Distinct
        \begin{enumerate}
            \item Group identical odd parts together.
            \item If an odd part $\displaystyle k$ appears $\displaystyle m$ times, write $\displaystyle m$ in binary:
                    $\displaystyle m = 2^{a_{1}} + 2^{a_{2}} + \cdots$ with distinct powers of two.
            \item Replace the $\displaystyle m$ copies of $\displaystyle k$ by $\displaystyle 2^{a_{1}}\cdot k,\;
                    2^{a_{2}}\cdot k,\; \cdots$
        \end{enumerate}
    \item Distinct $\rightarrow$ Odd
        \begin{enumerate}
            \item Factor each part as $\displaystyle 2^{a}\times \text{odd}$.
            \item Split it into $\displaystyle 2^{a}$ copies of that odd factor.
            \item Do this for all parts; now all parts are odd.
            \item Sort the resulting parts to obtain a partition into odd parts.
        \end{enumerate}
\end{itemize}
As an example:
\begin{equation}
    \begin{aligned}
        3 + 3 + 3 + 3 + 3 + 1 + 1
        &= 5\times 3 + 2\times 1 \\
        &= \left(2^{2} + 2^{0}\right)\times 3 + 2^{1}\times 1 \\
        &= 4\times 3 + 1\times 3 + 2\times 1 \\
        &= 12 + 3 + 2
    \end{aligned}
\end{equation}
\begin{equation}
    \begin{aligned}
        12 + 3 + 2
        &= 4\times 3 + 3 + 2\times 1 \\
        &= \left(3 + 3 + 3 + 3\right) + 3 + \left(1 + 1\right) \\
        &= 3 + 3 + 3 + 3 + 3 + 1 + 1
    \end{aligned}
\end{equation}

$\rhd$ \textbf{I.19.} \textit{Euler’s pentagonal number theorem}.
To find a recurrence for the partition numbers, we consider the reciprocal of the generating function $\displaystyle P\left(z\right)$:
\begin{equation}
    A\left(z\right) = \prod_{n = 1}^{\infty}\left(1 - z^{n}\right)
\end{equation}
Euler's Pentagonal Number Theorem states that this product expands into the following sparse power series:
\begin{equation}
    \begin{aligned}
        A\left(z\right)
            &= \left(\sum_{k = -\infty}^{\infty}\left(-1\right)^{k}z^{k\left(3k + 1\right)/2}\right) \\
            &= \left(\sum_{k = -\infty}^{\infty}\left(-1\right)^{k}z^{k\left(3k - 1\right)/2}\right) \\
            &= 1 - z^{1} - z^{2} + z^{5} + z^{7} - z^{12} - z^{15} + \cdots
    \end{aligned}
\end{equation}
The exponents $\displaystyle g_{k} = \frac{k\left(3k - 1\right)}{2}$ are the generalized pentagonal numbers.
The product of $\displaystyle P\left(z\right)$ and $\displaystyle A\left(z\right)$ must equal $\displaystyle 1$:
\begin{equation}
    \left(\sum_{n = 0}^{\infty}p\left(n\right)z^{n}\right)
        \left(\sum_{k = -\infty}^{\infty}\left(-1\right)^{k}z^{g_{k}}\right) = 1
\end{equation}
Comparing the coefficient of $\displaystyle z^{n}$ on both sides yields the algorithmically useful recurrence:
\begin{equation}
    p\left(n\right) = p\left(n - 1\right) + p\left(n - 2\right)
        - p\left(n - 5\right) - p\left(n - 7\right)
        + p\left(n - 12\right) + p\left(n - 15\right)\cdots
\end{equation}
A Python implementation is provided below:
\newpage
\begin{lstlisting}
    def compute_partitions(N):
    P = [0] * (N + 1)
    P[0] = 1  # Base case: p(0) = 1

    for i in range(1, N + 1):
        k = 1
        while True:
            # Calculate pentagonal numbers for k and -k
            g_k_pos = k * (3 * k - 1) // 2
            g_k_neg = k * (3 * k + 1) // 2

            # Determine sign: +1 for k=1,2 (terms 1,2), -1 for k=3,4 (terms 3,4)...
            # Simplest way: (-1)^(k-1)
            sign = 1 if k % 2 == 1 else -1

            # Add p(i - g_k_pos)
            if g_k_pos <= i:
                P[i] += sign * P[i - g_k_pos]
            else:
                break # No more terms will fit

            # Add p(i - g_k_neg)
            if g_k_neg <= i:
                P[i] += sign * P[i - g_k_neg]
            else:
                break

            k += 1
    return P
\end{lstlisting}

$\rhd$ \textbf{I.42.} \textit{Faster determination of Schröder numbers}.
\begin{equation}
    S\left(z\right) = \frac{1}{4}\left(1 + z - \sqrt{1 - 6z + z^{2}}\right)
\end{equation}
Consider in general
\begin{equation}
    p\left(z\right)S\left(z\right) + q\left(z\right) = \sqrt[k]{r\left(z\right)}
\end{equation}
where $\displaystyle p\left(z\right)$, $\displaystyle q\left(z\right)$, and $\displaystyle r\left(z\right)$ are functions of $\displaystyle z$.
Taking the derivative with respect to $\displaystyle z$,
\begin{equation}
    p'S + pS' + q' = \frac{1}{k}{r}^{\frac{1}{k}-1}r'
\end{equation}
Multiplying both sides by $\displaystyle r\left(z\right)$,
\begin{equation}
    p'rS + prS' + q'r = \frac{1}{k}{r}^{\frac{1}{k}}r' = \frac{1}{k}\left(pS + q\right)r'
\end{equation}
Rearranging,
\begin{equation}
    prS' + \left(p'r - \frac{1}{k}pr'\right)S = \frac{1}{k}qr' - q'r
\end{equation}
Plugging $\displaystyle p\left(z\right) = -4$, $\displaystyle q\left(z\right) = 1 + z$, $\displaystyle k = 2$,
and $\displaystyle r\left(z\right) = 1 - 6z + z^{2}$ into the equation and simplifying yields
\begin{equation}
    \left(z^{2} - 6z + 1\right)S' + \left(3 - z\right)S = 1 - z
\end{equation}
from which the recurrence follows.
