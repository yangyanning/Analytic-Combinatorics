\section{Generating functions as analytic objects}
% Note IV.6.
\noteblock{IV.6}{Integrals of rational fractions.}

We compute the two displayed integrals by closing the contour in the upper half-plane. The
only work is then residue computation plus a little finite algebra.

\begin{equation*}
    J_{m} = \int_{-\infty}^{\infty} \frac{\dd x}{(1 + x^{2})^{m}}
\end{equation*}
Set
\begin{equation*}
    f(z) = \frac{1}{(1 + z^{2})^{m}} = \frac{1}{(z - i)^{m}(z + i)^{m}}.
\end{equation*}
Integrate over the contour $\Gamma_{R}$ formed by the real interval $[-R, R]$ and the upper semicircle
$C_{R} = \{ z = Re^{i\theta}: 0 \leq \theta \leq \pi \}$.

For $z \in C_{R}$ and $R > 1$, $|z^{2} + 1| \geq |z^{2}| - 1 = R^{2} - 1$. Thus
\begin{equation*}
    |f(z)| \leq \frac{1}{(R^{2} - 1)^{m}}
\end{equation*}
The length of $C_{R}$ is $\pi R$, so
\begin{equation*}
    \left|\int_{C_{R}}f(z)\dd z\right| \leq \frac{\pi R}{(R^{2} - 1)^{m}} \to 0 \; \text{as} \; R \to \infty
\end{equation*}
So the contour integral reduces to a residue computation at the pole in the upper half-plane. Therefore
\begin{equation*}
    \lim_{R \to \infty} \int_{\Gamma_{R}}f(z)\dd z = \int_{-\infty}^{\infty} \frac{\dd x}{(1 + x^{2})^{m}} = J_{m}
\end{equation*}
For $R > 1$, the only singularity inside $\Gamma_{R}$ is the pole of order $m$ at $z = i$. The residue theorem then gives:
\begin{equation*}
    \int_{\Gamma_{R}}f(z)\dd z = 2\pi i\Res(f; i)
\end{equation*}
Now compute the residue at $z = i$:
\begin{equation*}
    \Res(f; i) = \frac{1}{(m - 1)!} \lim_{z \to i} \frac{\dd^{m - 1}}{\dd z^{m - 1}}[(z - i)^{m}f(z)]
        = \frac{1}{(m - 1)!} \lim_{z \to i} \frac{\dd^{m - 1}}{\dd z^{m - 1}}(z + i)^{-m}
\end{equation*}
The $k$th derivative of $g(z) \coloneqq (z + i)^{-m}$ is:
\begin{equation*}
    g^{(k)}(z) = (-1)^{k} \frac{(m + k - 1)!}{(m - 1)!}(z + i)^{-m - k}
\end{equation*}
For $k = m - 1$:
\begin{equation*}
    g^{(m - 1)}(z) = (-1)^{m - 1} \frac{(2m - 2)!}{(m - 1)!}(z + i)^{-2m + 1}
\end{equation*}
Evaluating at $z = i$ gives
\begin{equation*}
    g^{(m - 1)}(i) = (-1)^{m - 1} \frac{(2m - 2)!}{(m - 1)!}(2i)^{-2m + 1}
\end{equation*}
Therefore,
\begin{equation*}
    \begin{aligned}
        \Res(f; i)
        &= \frac{1}{(m - 1)!}(-1)^{m - 1} \frac{(2m - 2)!}{(m - 1)!}(2i)^{-2m + 1} \\
        &= (-1)^{m - 1} \frac{(2m - 2)!}{[(m - 1)!]^{2}} \frac{(-1)^{m}i}{2^{2m - 1}} \\
        &= -i \frac{(2m - 2)!}{[(m - 1)!]^{2}2^{2m - 1}} \\
    \end{aligned}
\end{equation*}
\begin{equation} \label{eq:IV.6.1}
    J_{m} = 2\pi i\left(-i \frac{(2m - 2)!}{[(m - 1)!]^{2}2^{2m - 1}}\right) = \frac{\pi}{2^{2m - 2}} \binom{2m - 2}{m - 1}
        = \frac{\pi}{4^{m - 1}} \binom{2m - 2}{m - 1}
\end{equation}

\partsep
\begin{equation*}
    K_{m} = \int_{-\infty}^{\infty} \frac{\dd x}{(1^{2} + x^{2})(2^{2} + x^{2}) \cdots (m^{2} + x^{2})}
\end{equation*}
Now set
\begin{equation*}
    f(z) = \frac{1}{\prod_{k = 1}^{m}(z^{2} + k^{2})}.
\end{equation*}
We use the same contour $\Gamma_{R}$.

For $R > m$, the poles inside $\Gamma_{R}$ are $z_{k} = ik, k = 1, \cdots m$. On $C_{R}$, $|z^{2} + k^{2}| \geq R^{2} - k^{2} \geq (R/2)^{2}$ eventually,
so $|f(z)| \leq \text{const}/R^{2m}$ while the length of $C_{R}$ is $\pi R$. Hence,
\begin{equation*}
    \lim_{R \to \infty} \int_{C_{R}}f(z)\dd z = 0
\end{equation*}
and the residue theorem yields
\begin{equation*}
    K_{m} = 2\pi i \sum_{k = 1}^{m} \Res(f; ik)
\end{equation*}
Since the poles are simple,
\begin{equation*}
    \Res(f; ik) = \lim_{z \to ik}(z - ik)f(z) = \frac{1}{2ik} \prod_{j \neq k} \frac{1}{(ik)^{2} + j^{2}}
        = \frac{1}{2ik} \prod_{j \neq k} \frac{1}{j^{2} - k^{2}}
\end{equation*}
Therefore,
\begin{equation*}
    K_{m} = 2\pi i \sum_{k = 1}^{m} \frac{1}{2ik} \prod_{j \neq k} \frac{1}{j^{2} - k^{2}}
        = \pi \sum_{k = 1}^{m} \frac{1}{k} \prod_{j \neq k} \frac{1}{j^{2} - k^{2}}
\end{equation*}
The analytic work is now done; what remains is straightforward algebra, though it takes a little patience.
Factor the product:
\begin{equation*}
    \prod_{j \neq k}(j^{2} - k^{2}) = \prod_{j \neq k}(j - k)\prod_{j \neq k}(j + k).
\end{equation*}
For fixed $k$,
\begin{equation*}
    \prod_{j \neq k}(j - k) = (1 - k)(2 - k) \cdots ((k - 1) - k)(1)(2) \cdots (m - k) = (-1)^{k - 1}(k - 1)!(m - k)!
\end{equation*}
\begin{equation*}
    \prod_{j \neq k}(j + k) = (1 + k)(2 + k) \cdots ((k - 1) + k)(2k + 1)(2k + 2) \cdots (m + k) = \frac{(m + k)!}{k!(2k)}
\end{equation*}
Therefore,
\begin{equation*}
    \frac{1}{k} \prod_{j \neq k} \frac{1}{j^{2} - k^{2}} = \frac{1}{k} \left((-1)^{k - 1}(k - 1)!(m - k)! \cdot \frac{(m + k)!}{k!(2k)} \right)^{-1}
        = \frac{(-1)^{k - 1}2k}{(m - k)!(m + k)!}
\end{equation*}
and
\begin{equation*}
    K_{m} = \pi \sum_{k = 1}^{m} \frac{(-1)^{k - 1}2k}{(m - k)!(m + k)!} = \frac{2\pi}{(2m)!} \sum_{k = 1}^{m}(-1)^{k - 1}k \binom{2m}{m - k}
\end{equation*}

To simplify the sum
\begin{equation*}
    S \coloneqq \sum_{k = 1}^{m}(-1)^{k - 1}k \binom{2m}{m - k},
\end{equation*}
generating functions are again the quickest tool. The aim is to turn the alternating sum into a single coefficient extraction. Recall that the coefficient of $z^{m}$ in a product $A(z)B(z)$ is
\begin{equation*}
    [z^{m}] A(z)B(z) = \sum_{k} a_{k}b_{m - k}
\end{equation*}
Take
\begin{equation*}
    A(z) = \sum_{k = 1}^{m}(-1)^{k - 1}kz^{k},
    \qquad
    B(z) = (1 + z)^{2m} = \sum_{j = 0}^{2m} \binom{2m}{j} z^{j}.
\end{equation*}
Then the coefficient of $z^{m}$ in $A(z)B(z)$ is exactly $S$:
\begin{equation} \label{eq:IV.6.2}
    S = [z^{m}] \sum_{k = 1}^{m}(-1)^{k - 1}kz^{k}(1 + z)^{2m}
\end{equation}
Since we only care about the coefficient of $z^{m}$, terms of degree $> m$ in $A(z)$ are irrelevant. So we may extend the sum to infinity without changing the coefficient:
\begin{equation*}
    A(z) \equiv \sum_{k = 1}^{\infty}(-1)^{k - 1}kz^{k} = \frac{z}{(1 + z)^{2}} \pmod{z^{m + 1}}.
\end{equation*}
Hence, by \eqref{eq:IV.6.2},
\begin{equation*}
    S = [z^{m}] \frac{z}{(1 + z)^{2}} \cdot (1 + z)^{2m} = [z^{m}] z(1 + z)^{2m - 2} = \binom{2m - 2}{m - 1}
\end{equation*}

Finally,
\begin{equation} \label{eq:IV.6.3}
    K_{m} = \frac{2\pi}{(2m)!} \binom{2m - 2}{m - 1} = \frac{\pi}{m(2m - 1)[(m - 1)!]^{2}}
\end{equation}
