% Note II.19.
\noteblock{II.19}{Labelled hierarchies.}

The class $\mathcal{L}$ of labelled hierarchies is made of rooted trees whose leaves carry
labels, while internal nodes are unlabelled and have outdegree at least $2$. The size is the
number of labels, so internal nodes do not contribute to size. This gives the labelled
specification
\begin{equation*}
    \mathcal{L} = \mathcal{Z} + \mathrm{SET}_{\geq 2}(\mathcal{L}),
\end{equation*}
and therefore
\begin{equation*}
    L(z) = z + e^{L(z)} - 1 - L(z).
\end{equation*}
The term $\mathcal{Z}$ is the one-leaf hierarchy. The term
$\mathrm{SET}_{\geq 2}(\mathcal{L})$ says that an internal node is just an unordered set of
at least two smaller hierarchies.

The book notes that this equation can be solved in terms of the Cayley tree function $T$,
defined by $T(x) = xe^{T(x)}$. Let us derive the displayed form. First rewrite the equation as
\begin{equation*}
    e^{L(z)} = 2L(z) - z + 1.
\end{equation*}
Now set
\begin{equation*}
    u = \frac{z - 1}{2} - L(z).
\end{equation*}
Then $2L(z) - z + 1 = -2u$ and $-L(z) = u - (z - 1)/2$. Thus
\begin{equation*}
    1 = (2L(z) - z + 1)e^{-L(z)}
      = -2u\,e^{u - (z - 1)/2}.
\end{equation*}
Hence
\begin{equation*}
    u e^{u} = -\frac{1}{2}e^{(z - 1)/2}.
\end{equation*}
Since $W(y)e^{W(y)} = y$ and $T(x) = -W(-x)$, we get
\begin{equation*}
    u = W\left(-\frac{1}{2}e^{(z - 1)/2}\right)
      = -T\left(\frac{1}{2}e^{z/2 - 1/2}\right).
\end{equation*}
Substituting back gives
\begin{equation*}
    L(z) = T\left(\frac{1}{2} e^{z/2 - 1/2}\right) + \frac{z}{2} - \frac{1}{2}.
\end{equation*}
Expanding the EGF begins as
\begin{equation*}
    L(z) = z + \frac{z^{2}}{2!} + 4\frac{z^{3}}{3!}
        + 26\frac{z^{4}}{4!} + 236\frac{z^{5}}{5!} + \cdots.
\end{equation*}
So the labelled counts start as
\begin{equation*}
    L_{0}, L_{1}, L_{2}, L_{3}, L_{4}, L_{5}, \dots
    = 0, 1, 1, 4, 26, 236, \dots.
\end{equation*}

% II.5.2.
\topicblock{II.5.2}{Mappings and associated graphs.}

Let $\mathcal{T}$ be the class of all Cayley trees, $\mathcal{K}$ the class of connected
functional graphs, and $\mathcal{F}$ the class of all mappings. The symbolic decomposition is
\begin{equation*}
    \mathcal{F} = \mathrm{SET}(\mathcal{K}), \qquad
    \mathcal{K} = \mathrm{CYC}(\mathcal{T}), \qquad
    \mathcal{T} = \mathcal{Z} \star \mathrm{SET}(\mathcal{T}).
\end{equation*}
At the level of EGFs this becomes
\begin{equation*}
    F(z) = e^{K(z)}, \qquad
    K(z) = \log\frac{1}{1 - T(z)}, \qquad
    T(z) = ze^{T(z)}.
\end{equation*}
Thus
\begin{equation*}
    F(z) = \frac{1}{1 - T(z)}.
\end{equation*}
This is reassuring, since $F_n$ should count all mappings $[n]\to[n]$, hence $F_n=n^n$.
Here is the Lagrange-inversion check. For $T=z\phi(T)$ with $\phi(w)=e^w$,
\begin{equation*}
    [z^{n}] H(T(z)) = \frac{1}{n} [w^{n - 1}] H'(w)\phi(w)^{n}
\end{equation*}
and taking $H(w) = (1 - w)^{-1}$ gives
\begin{equation*}
    \begin{aligned}
        F_{n}
        &= n![z^{n}] F(z) \\
        &= n!\frac{1}{n} [w^{n - 1}] \left(\frac{1}{1 - w} \right)'(e^{w})^{n} \\
        &= (n - 1)![w^{n - 1}] \frac{e^{nw}}{(1 - w)^{2}} \\
        &= (n - 1)![w^{n - 1}] \sum_{k = 0}^{\infty}(k + 1)w^{k} \cdot \sum_{j = 0}^{\infty} \frac{n^{j}}{j!}w^{j} \\
        &= (n - 1)!\sum_{j = 0}^{n - 1}((n - 1 - j) + 1)\frac{n^{j}}{j!} \\
        &= (n - 1)!\sum_{j = 0}^{n - 1}(n - j)\frac{n^{j}}{j!} \\
        &= (n - 1)!\left(\sum_{j = 0}^{n - 1} n \frac{n^{j}}{j!} - \sum_{j = 0}^{n - 1} j \frac{n^{j}}{j!}\right) \\
        &= (n - 1)!\left(\sum_{j = 0}^{n - 1} \frac{n^{j + 1}}{j!} - \sum_{j = 1}^{n - 1} \frac{n^{j}}{(j - 1)!}\right) \\
        &= (n - 1)!\left(\sum_{j = 0}^{n - 1} \frac{n^{j + 1}}{j!} - \sum_{j = 0}^{n - 2} \frac{n^{j + 1}}{j!}\right) \\
        &= (n - 1)!\frac{n^{n}}{(n - 1)!} \\
        &= n^{n}
    \end{aligned}
\end{equation*}

For connected functional graphs, start from
\begin{equation*}
    K(z) = \log\frac{1}{1 - T(z)}
        = \sum_{k \geq 1} \frac{T(z)^{k}}{k}.
\end{equation*}
From Note II.18, for $1 \leq k \leq n$,
\begin{equation*}
    n![z^{n}] \frac{T(z)^{k}}{k!} = \binom{n - 1}{k - 1} n^{n - k} \implies
        n![z^{n}] \frac{T(z)^{k}}{k} = (k - 1)!\binom{n - 1}{k - 1} n^{n - k} = \frac{(n - 1)!}{(n - k)!} n^{n - k}
\end{equation*}
Therefore
\begin{equation*}
    \begin{aligned}
        K_{n}
        &= n![z^{n}] K(z) \\
        &= \sum_{k = 1}^{n} \frac{(n - 1)!}{(n - k)!} n^{n - k} \\
        &= n^{n - 1} \sum_{k = 1}^{n} \frac{(n - 1)!}{(n - k)!n^{k - 1}}.
    \end{aligned}
\end{equation*}
The last sum is Ramanujan's $Q$-function:
\begin{equation*}
    Q(n)
        = 1 + \frac{n - 1}{n}
            + \frac{(n - 1)(n - 2)}{n^{2}}
            + \cdots
            + \frac{(n - 1)!}{n^{n - 1}}.
\end{equation*}
Thus
\begin{equation*}
    K_n = n^{n - 1}Q(n).
\end{equation*}

\topicblock{II.5.2}{Constrained mappings.}

We only look at the first constrained class: mappings without fixed points. In the functional
graph decomposition, a fixed point is a $1$-cycle, with a rooted tree possibly attached to it. Since
\[
    \mathrm{CYC}(\mathcal{T}) \quad\leadsto\quad \log\frac{1}{1 - T(z)}
\]
and cycles of length $1$ contribute exactly $T(z)$, removing them changes the connected
component EGF to
\[
    \log\frac{1}{1 - T(z)} - T(z).
\]
Thus the EGF of all fixed-point-free mappings is
\begin{equation*}
    \exp\left(\log\frac{1}{1 - T(z)} - T(z)\right)
    = \frac{e^{-T(z)}}{1 - T(z)}.
\end{equation*}
We want to recover the direct count, namely the fact that each element has $n - 1$ possible images:
\begin{equation*}
    n![z^{n}] \frac{e^{-T(z)}}{1 - T(z)} = (n - 1)^{n}
\end{equation*}
by Lagrange inversion. For $n \geq 2$, with $H(w) = \frac{e^{-w}}{1 - w}$ and $\phi(w) = e^{w}$, we obtain
\begin{equation*}
    \begin{aligned}
        n![z^{n}] \frac{e^{-T(z)}}{1 - T(z)}
        &= n!\frac{1}{n} [w^{n - 1}] \left(\frac{e^{-w}}{1 - w}\right)'(e^{w})^{n} \\
        &= (n - 1)![w^{n - 1}] \frac{we^{(n - 1)w}}{(1 - w)^{2}} \\
        &= (n - 1)![w^{n - 2}] \frac{e^{(n - 1)w}}{(1 - w)^{2}} \\
        &= (n - 1)![w^{n - 2}] \sum_{k = 0}^{\infty}(k + 1)w^{k} \cdot \sum_{j = 0}^{\infty} \frac{(n - 1)^{j}}{j!}w^{j} \\
        &= (n - 1)!\sum_{j = 0}^{n - 2}((n - 2 - j) + 1)\frac{(n - 1)^{j}}{j!} \\
        &= (n - 1)!\sum_{j = 0}^{n - 2}(n - 1 - j)\frac{(n - 1)^{j}}{j!} \\
        &= (n - 1)!\left(
            \sum_{j = 0}^{n - 2}\frac{(n - 1)^{j + 1}}{j!}
            - \sum_{j = 1}^{n - 2}\frac{(n - 1)^{j}}{(j - 1)!}
        \right) \\
        &= (n - 1)!\left(
            \sum_{j = 0}^{n - 2}\frac{(n - 1)^{j + 1}}{j!}
            - \sum_{j = 0}^{n - 3}\frac{(n - 1)^{j + 1}}{j!}
        \right) \\
        &= (n - 1)!\frac{(n - 1)^{n - 1}}{(n - 2)!} \\
        &= (n - 1)^{n}
    \end{aligned}
\end{equation*}
For $n = 1$ the statement is also true, since there is no fixed-point-free map from a singleton
to itself.

\topicblock{II.5.3}{Unrooted trees and acyclic graphs.}

Let $\mathcal{U}$ be the class of unrooted labelled trees. A rooted labelled tree is just an
unrooted labelled tree with one vertex distinguished as the root. Therefore, for $n \geq 1$,
\begin{equation*}
    T_n = nU_n.
\end{equation*}
This immediately gives Cayley's unrooted-tree count
\begin{equation*}
    U_n = n^{n - 2},
\end{equation*}
but the generating-function form is also useful. Start from
\begin{equation*}
    T(z) = \sum_{n \geq 0} T_{n} \frac{z^{n}}{n!}, \quad
    U(z) = \sum_{n \geq 0} U_{n} \frac{z^{n}}{n!}
\end{equation*}
Differentiate $U(z)$ and multiply by $z$:
\begin{equation*}
    z \frac{\dd}{\dd z} U(z) = z \frac{\dd}{\dd z} \sum_{n \geq 0} U_{n} \frac{z^{n}}{n!} = z \sum_{n \geq 1} nU_{n} \frac{z^{n - 1}}{n!}
        = \sum_{n \geq 1} T_{n} \frac{z^{n}}{n!} = T(z)
\end{equation*}
So $\frac{\dd}{\dd z} U(z) = \frac{T(z)}{z}$. Since $U(0) = 0$, integrating gives
\begin{equation*}
    U(z) = \int_{0}^{z} \frac{T(w)}{w}\,\dd w.
\end{equation*}
Now evaluate the integral using the functional equation for rooted trees:
\begin{equation*}
    T(z) = ze^{T(z)}
\end{equation*}
Write $t = T(w)$. Then $w = te^{-t}$ and
\begin{equation*}
    \dd w = e^{-t}(1 - t)\,\dd t.
\end{equation*}
As $w$ runs from $0$ to $z$, $t$ runs from $0$ to $T(z)$, so
\begin{equation*}
    U(z)
        = \int_{0}^{T(z)} \frac{t}{te^{-t}} e^{-t}(1 - t)\,\dd t
        = \int_{0}^{T(z)} (1 - t)\,\dd t
        = T(z) - \frac{T(z)^{2}}{2}.
\end{equation*}
Thus the EGF of all acyclic labelled graphs, i.e. sets of unrooted labelled trees, is
\begin{equation*}
    A(z) = e^{U(z)} = \exp\left(T(z) - \frac{T(z)^2}{2}\right).
\end{equation*}

% Note II.22.
\noteblock{II.22}{2–Regular graphs.}

We only prove the equinumerosity statement. Take $n$ labelled lines
$L_{1},\ldots,L_{n}$ in general position: no two lines are parallel and no three lines meet at
one point. For the reverse direction below, we also use the usual generic reading of the setup:
apart from the original lines themselves, no line contains three of the intersection points.
Now turn a cloud into a graph on vertex set $\{1,\ldots,n\}$:
\begin{itemize}
    \item The line $L_i$ becomes vertex $i$.
    \item The chosen intersection point $L_i \cap L_j$ becomes the edge $\{i,j\}$.
\end{itemize}
This is already reversible: an edge set tells us exactly which intersections to choose.

It remains to check the condition. Let $d_i$ be the number of chosen intersection points lying
on $L_i$; in the graph this is exactly the degree of vertex $i$. Since a cloud contains $n$
intersection points and each chosen point lies on two lines,
\begin{equation*}
    \sum_{i = 1}^{n} d_i = 2n.
\end{equation*}
If the cloud has no three collinear points, then no line $L_i$ contains three chosen points, so
$d_i \leq 2$ for every $i$. The displayed sum then forces $d_i = 2$ for every $i$, hence the
associated graph is $2$-regular.

Conversely, if the graph is $2$-regular, then $d_i = 2$ for every $i$. Thus no original line
contains three chosen intersection points; by the genericity assumption, this means the
chosen intersections form a cloud. Therefore clouds on $n$ labelled lines and $2$-regular
graphs on $n$ labelled vertices are equinumerous.

% Note II.23.
\noteblock{II.23}{Wright’s surgery.}

Let $W_k$ be the EGF of connected graphs of excess $k$, and introduce an edge marker by
\begin{equation*}
    w_k(z,y) = y^k W_k(zy).
\end{equation*}
This is just a compact way to say that a graph with $n$ vertices and excess $k$ has
$n+k$ edges: each vertex contributes one factor $zy$, and the extra $k$ edges contribute
$y^k$.

Start with a connected graph of excess $k+1$. Choose one edge, orient it, and remove it.
In a generating function with $y$ marking edges, $\partial_y$ points an edge and removes one
factor of $y$; in this surgery picture, this is exactly the chosen edge that we delete. The
factor $2$ chooses its orientation, so the left-hand side is
\begin{equation*}
    2\partial_y w_{k+1}.
\end{equation*}
After the edge is removed, the total excess drops by $1$, from $k+1$ to $k$. There are two
possibilities.

\textbf{Case 1: the graph stays connected.}

If the removed edge was not a bridge, the remaining graph is connected of excess $k$, with two
ordered distinguished vertices: the former tail and head of the removed directed edge. Choosing
an ordered pair of distinct vertices in a labelled graph is represented by
\begin{equation*}
    z^2\partial_z^2 w_k.
\end{equation*}
However, not every ordered pair is allowed. If the two chosen vertices are already joined by an
edge, adding back the deleted edge would create a multiple edge, which is forbidden for simple
graphs. Ordered adjacent pairs are counted by choosing an edge and orienting it, namely
$2y\partial_y w_k$. Thus the connected case contributes
\begin{equation*}
    z^2\partial_z^2 w_k - 2y\partial_y w_k.
\end{equation*}

\textbf{Case 2: the graph splits into two components.}

If the removed edge was a bridge, the remaining graph has two connected components. Suppose
the component containing the tail has excess $h$; then the other component has excess $k-h$,
because the total remaining excess is $k$. We must mark one vertex in each component, so the
contribution is
\begin{equation*}
    \left(z\partial_z w_h\right)\left(z\partial_z w_{k-h}\right).
\end{equation*}
The possible excess values start at $-1$, since trees have excess $-1$. Hence $h$ ranges from
$-1$ to $k+1$, which also keeps $k-h \geq -1$.

Putting the connected and disconnected cases together gives Wright's differential recurrence:
\begin{equation*}
    2\partial_{y}w_{k + 1} = \left(z^{2}\partial_{z}^{2}w_{k} - 2y\partial_{y}w_{k}\right) +
        \sum_{h = -1}^{k + 1} \left(z\partial_{z}w_{h}\right) \cdot \left(z\partial_{z}w_{k - h}\right)
\end{equation*}
The same recurrence can then be specialized to $W_k(z)$ by setting $y=1$.

\noteblock{II.28}{Do two permutations generate the symmetric group?}

The index in the book appears to have a typo here: the relation should involve $I_{n+1}$,
not $I_n$. We prove
\begin{equation*}
    n!^{2}\pi_{n} = (n - 1)!I_{n + 1}
\end{equation*}
where $I_m$ denotes the number of indecomposable permutations of size $m$.

The first useful observation is that the EGF of pairs of permutations is the same formal series
as the ordinary generating function of permutations:
\begin{equation*}
    \sum_{n \geq 0} n!^{2} \frac{z^{n}}{n!} = \sum_{n \geq 0} n!z^{n},
\end{equation*}
so write
\begin{equation*}
    P(z) = \sum_{n \geq 0} n!z^n.
\end{equation*}
By the exponential formula, the EGF $C(z)$ of connected pairs satisfies
\begin{equation*}
    C(z) = \log P(z).
\end{equation*}
Since every permutation decomposes uniquely into a sequence of indecomposable blocks, the generating function $I(z)$ of indecomposable permutations satisfies
\begin{equation*}
    I(z) = \sum_{n \geq 1} I_n z^n, \qquad
    P(z) = \frac{1}{1 - I(z)}, \qquad
    I(z) = 1 - \frac{1}{P(z)}.
\end{equation*}
Now differentiate and extract the consequence:
\begin{equation*}
    C'(z) = \frac{P'(z)}{P(z)} = P'(z)(1 - I(z)) = P'(z) - P'(z)I(z)
\end{equation*}
Next we extract the coefficient of $z^{n - 1}$ from both sides.

\stephead{1}{Extract the left-hand side coefficient.}
Since $C(z) = \sum_{n \geq 1}(n!^{2}\pi_{n})\frac{z^{n}}{n!}$, its derivative is:
\begin{equation*}
    C'(z) = \sum_{n \geq 1}(n!^{2}\pi_{n})\frac{z^{n - 1}}{(n - 1)!}
\end{equation*}
So the coefficient of $z^{n - 1}$ is:
\begin{equation*}
    [z^{n - 1}] C'(z) = \frac{n!^{2}\pi_{n}}{(n - 1)!} = n \cdot n!\pi_{n}
\end{equation*}

\stephead{2}{Extract the right-hand side coefficient.}
First simplify the right-hand side. Differentiating $P(z)(1 - I(z)) = 1$ gives
\begin{equation*}
    P'(z)(1 - I(z)) - P(z)I'(z) = 0 \implies P'(z) - P'(z)I(z) = P(z)I'(z)
\end{equation*}
Therefore
\begin{equation*}
    [z^{n - 1}] (P'(z) - P'(z)I(z)) = [z^{n - 1}] P(z)I'(z) = \sum_{k = 1}^{n} kI_{k}(n - k)!
\end{equation*}
Call this last sum $S_n$. We now show directly that $S_n=I_{n+1}$.

The only recurrence we need comes from $P(z)=1+I(z)P(z)$. Extracting $[z^N]$ for
$N\geq 1$ gives
\begin{equation} \label{eq:II.28.1}
    \sum_{k = 1}^{N} I_{k}(N - k)! = N!
\end{equation}
Now apply this with $N=n+1$, and separate the last term:
\begin{equation} \label{eq:II.28.2}
    \sum_{k = 1}^{n} I_{k}(n + 1 - k)! = (n + 1)! - I_{n + 1}.
\end{equation}
On the other hand, rewrite the same left-hand side by pulling out the factor
$n+1-k$:
\begin{equation*}
    \begin{aligned}
        \sum_{k = 1}^{n} I_{k}(n + 1 - k)!
        &= \sum_{k = 1}^{n} I_{k}(n + 1 - k)(n - k)! \\
        &= (n + 1)\sum_{k = 1}^{n} I_{k}(n - k)! - \sum_{k = 1}^{n} kI_{k}(n - k)! \\
    \end{aligned}
\end{equation*}
The first sum is $n!$ by \eqref{eq:II.28.1} with $N=n$, and the second sum is exactly
$S_n$. Hence
\begin{equation*}
    \sum_{k = 1}^{n} I_{k}(n + 1 - k)! = (n + 1)! - S_n.
\end{equation*}
Comparing this with \eqref{eq:II.28.2} gives $S_n=I_{n+1}$.

\stephead{3}{Conclude the identity.}
Finally, $n \cdot n!\pi_{n} = S_{n}$, so
\begin{equation*}
    n!^{2}\pi_{n} = (n - 1)!I_{n + 1}.
\end{equation*}
